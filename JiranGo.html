function openPayment() { // 1. 获取通用必填项 const isBuy = (currentType === 'buy'); if (isBuy) { // --- 帮忙买验证 --- const addr = document.getElementById('buy-address').value.trim(); const items = document.querySelectorAll('#buy-items .item-name'); // 检查地址 if (!addr) return jiranAlert("请填好送货地址哦！不然跑腿员不知道要把东西送去哪家 🏠"); // 检查是否至少填了一个物品名称 let hasItem = false; items.forEach(input => { if(input.value.trim()) hasItem = true; }); if (!hasItem) return jiranAlert("清单还是空的哦！请告诉我们要帮你买什么东西 🛒"); } else { // --- 帮忙送验证 --- const pickup = document.getElementById('send-pickup').value.trim(); const dropoff = document.getElementById('send-dropoff').value.trim(); const items = document.querySelectorAll('#send-items .item-name'); // 检查取货和送货地址 if (!pickup) return jiranAlert("请填写【哪里取货】！我们需要知道去哪里拿包裹 📦"); if (!dropoff) return jiranAlert("请填写【送到哪里】！目的地不能为空哦 🚗"); // 检查物品描述 let hasItem = false; items.forEach(input => { if(input.value.trim()) hasItem = true; }); if (!hasItem) return jiranAlert("请简单描述一下是什么货品（例：面包、包裹），方便我们安排车位 📦"); } // 2. 所有检查通过，才显示付款弹窗 document.getElementById('pay-modal').style.display = 'flex'; }
